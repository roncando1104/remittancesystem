CREATE TABLE REM_CUSTOMER
(REFERENCE_NUMBER VARCHAR(20) PRIMARY KEY,
NAME VARCHAR(50) NOT NULL,
ID_TYPE VARCHAR(20) NOT NULL,
MOBILE_NUMBER VARCHAR(15) NOT NULL,
PURPOSE VARCHAR(50) NOT NULL,
RELATIONSHIP_TO_THE_RECEIVER VARCHAR(20) NOT NULL,
ADDRESS VARCHAR(100) NOT NULL,
ISSUE_DATE DATE NOT NULL,
ISSUE_PLACE VARCHAR(30) NOT NULL,
EXPIRY_DATE DATE NOT NULL,
AMOUNT VARCHAR(10) NOT NULL,
CHARGE VARCHAR(10) NOT NULL,
TOTAL_AMOUNT VARCHAR(10) NOT NULL)


CREATE TABLE REM_BENEFICIARIES
(REFERENCE_NUMBER VARCHAR(20) PRIMARY KEY,
NAME VARCHAR(50) NOT NULL,
ID_TYPE VARCHAR(20) NOT NULL,
MOBILE_NUMBER VARCHAR(15) NOT NULL,
ADDRESS VARCHAR(100) NOT NULL,
RELATIONSHIP_TO_THE_SENDER VARCHAR(20) NOT NULL,
CORRESPONDENT VARCHAR(40) NOT NULL,
BRANCH_ADDRESS VARCHAR(50) NOT NULL,
AMOUNT_RECEIVED VARCHAR(10) NOT NULL)

CREATE TABLE REM_STATUS
(REFERENCE_NUMBER VARCHAR(20) PRIMARY KEY,
EMPLOYEE_ID VARCHAR(10) NOT NULL,
ISSUE_DATE DATE NOT NULL,
EXPIRY_DATE DATE NOT NULL,
STATUS VARCHAR(20),
AMOUNT VARCHAR(20) NOT NULL)
